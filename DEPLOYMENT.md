```bash

#   Remove anonymous users:      YES
```

```sql
-- Run as root: mysql -u root -p

GRANT ALL PRIVILEGES ON beeliin.* TO 'beeliin'@'localhost';
FLUSH PRIVILEGES;
EXIT;
```

## 5. Redis Setup

```bash
apt install -y redis-server
```

Edit `/etc/redis/redis.conf`:

```ini
# Bind to localhost only — never expose Redis to the public internet
bind 127.0.0.1 ::1

# Require a password
requirepass REDIS_PASSWORD_HERE

# Persist to disk (optional but recommended)
appendonly yes
```

```bash
systemctl enable redis-server
systemctl restart redis-server
redis-cli -a REDIS_PASSWORD_HERE ping   # should return PONG
```

## 8. Nginx Configuration

```bash
apt install -y nginx
systemctl enable nginx
```

Create `/etc/nginx/sites-available/beeliin-api`:

```nginx
server {
    listen 80;
    server_name api.yourdomain.com;

    # Redirect all HTTP to HTTPS — Certbot will update this block after SSL setup
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name api.yourdomain.com;

    root /var/www/beeliin-api/public;
    index index.php;

    # SSL certificates — managed by Certbot (see Section 9)
    ssl_certificate     /etc/letsencrypt/live/api.yourdomain.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/api.yourdomain.com/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;

    # Security headers
    add_header X-Frame-Options           "SAMEORIGIN"   always;
    add_header X-Content-Type-Options    "nosniff"      always;
    add_header X-XSS-Protection          "1; mode=block" always;
    add_header Referrer-Policy           "strict-origin-when-cross-origin" always;
    add_header Strict-Transport-Security "max-age=63072000; includeSubDomains; preload" always;

    # Stripe webhooks — must not be buffered (signature is time-sensitive)
    location = /api/payments/webhook/stripe {
        fastcgi_pass unix:/run/php/php8.3-fpm.sock;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include       fastcgi_params;
        fastcgi_read_timeout 30;
    }

    # Health check — low timeout, no logging
    location = /api/health {
        access_log off;
        fastcgi_pass unix:/run/php/php8.3-fpm.sock;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include       fastcgi_params;
    }

    # All other requests
    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }

    location ~ \.php$ {
        fastcgi_pass        unix:/run/php/php8.3-fpm.sock;
        fastcgi_index       index.php;
        fastcgi_param       SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include             fastcgi_params;
        fastcgi_hide_header X-Powered-By;
        fastcgi_read_timeout 60;
    }

    location ~ /\.(?!well-known).* {
        deny all;
    }

    # Block access to sensitive files
    location ~* \.(env|log|key|pem)$ {
        deny all;
    }

    client_max_body_size 20M;

    access_log /var/log/nginx/beeliin-api.access.log;
    error_log  /var/log/nginx/beeliin-api.error.log warn;
}
```

```bash
ln -s /etc/nginx/sites-available/beeliin-api /etc/nginx/sites-enabled/
nginx -t                  # must print: syntax is ok
systemctl reload nginx
```

## 10. Supervisor Queue Workers

Queue workers process jobs from `ProvisionServer`, billing automation, and renewal generation.

```bash
apt install -y supervisor
systemctl enable supervisor
```

Create `/etc/supervisor/conf.d/beeliin-worker.conf`:

```ini
[program:beeliin-worker]
process_name=%(program_name)s_%(process_num)02d
command=php /var/www/beeliin-api/artisan queue:work redis \
    --sleep=3 \
    --tries=3 \
    --max-time=3600 \
    --queue=default
autostart=true
autorestart=true
stopasgroup=true
killasgroup=true
user=www-data
numprocs=3
redirect_stderr=true
stdout_logfile=/var/log/supervisor/beeliin-worker.log
stdout_logfile_maxbytes=10MB
stdout_logfile_backups=5
stopwaitsecs=60
```

```bash
supervisorctl reread
supervisorctl update
supervisorctl start beeliin-worker:*
supervisorctl status          # all three workers should show RUNNING
```

## 11. Deployment Commands

Run these on every new release. SSH into the server as the `deploy` user.

```bash
cd /var/www/beeliin-api

# 1. Pull latest code
git pull origin master

# 2. Install/update dependencies (no dev packages in production)
composer install --no-dev --optimize-autoloader

# 3. Run any new migrations
php artisan migrate --force

# 4. Rebuild caches
php artisan config:cache
php artisan route:cache
php artisan view:cache
php artisan event:cache

# 5. Restart queue workers so they pick up new code
supervisorctl restart beeliin-worker:*

# 6. Reload PHP-FPM (zero-downtime — graceful)
systemctl reload php8.3-fpm
```

### One-liner alias (optional)

Add to `/home/deploy/.bashrc`:

```bash
alias deploy-api='cd /var/www/beeliin-api && git pull origin master && composer install --no-dev --optimize-autoloader && php artisan migrate --force && php artisan config:cache && php artisan route:cache && php artisan view:cache && supervisorctl restart beeliin-worker:* && systemctl reload php8.3-fpm && echo "Deploy complete."'
```

## 12. Rollback Procedure

### Quick rollback to the previous commit

```bash
cd /var/www/beeliin-api

# Find the commit hash to roll back to
git log --oneline -10

# Roll back application code
git checkout <previous-commit-hash>

# Rebuild caches for the restored code
php artisan config:cache
php artisan route:cache
php artisan view:cache

# Restart workers
supervisorctl restart beeliin-worker:*
systemctl reload php8.3-fpm
```

### Rollback a migration (if the new release added one)

```bash
# Roll back the last migration batch only
php artisan migrate:rollback --step=1 --force

# Then check the current migration state
php artisan migrate:status
```

### Full rollback checklist

| Step | Command |
|---|---|
| Revert code | `git checkout <hash>` |
| Revert migration | `php artisan migrate:rollback --step=1 --force` |
| Rebuild caches | `php artisan config:cache && php artisan route:cache && php artisan view:cache` |
| Restart workers | `supervisorctl restart beeliin-worker:*` |
| Reload PHP-FPM | `systemctl reload php8.3-fpm` |
| Verify health | `curl https://api.yourdomain.com/api/health` |

> After rollback, tag the broken commit and open an incident before re-deploying.
